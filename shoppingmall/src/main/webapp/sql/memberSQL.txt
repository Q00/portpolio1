create table Member{
	ID VARCHAR2(20),
	PASSWD VARCHAR2(15),
	NAME VARCHAR2(20),
	SSN1 NUMBER,
	SSN2 NUMBER,
	EMAIL VARCHAR2(25),
	PH1 NUM(4),
	PH2 NUM(5),
	PH2 NUM(5),
	ZIPCODE VARCHAR2(13),
	ADDR1 VARCHAR2(70),
	ADDR2 VARCHAR2(70),
	MEMBER_ADMIN NUMBER,
	MEMBER_JOIN_DATE DATE
};

CREATE TABLE ZIPCODE (
  ID NUMBER,
  ZIPCODE VARCHAR2(7),
  SIDO VARCHAR2(10),
  GUGUN VARCHAR2(30),
  DONG VARCHAR2(36),
  RI VARCHAR2(60),
  BUNJI VARCHAR2(50),
  PRIMARY KEY (ID)
);
